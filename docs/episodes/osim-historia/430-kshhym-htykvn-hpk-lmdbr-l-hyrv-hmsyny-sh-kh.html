האמירה שתוכנית אטלנטרופה הייתה אתגר טכנולוגי אדיר היא כמובן לשון המעטה. מדובר במיזם התעשייתי הגדול ביותר שידעה האנושות מאז ומעולם. סורגל לא היה תמים. הוא ידע שמימוש חזונו יארך מאות שנים ויעלה סכומי כסף בלתי נתפסים, אבל הוא עשה את תוכניותיו בהתאם. הוא פרס את מיזמי הבנייה על פני עשרות שנים כך שלא יהיה צורך להקים את כולם בבת אחת וניתן יהיה להשתמש ברווחים מהסכר ההידרואלקטרי כדי לממן את שאר הפרויקטים. הוא לקח דוגמה ממיזמים טכנולוגיים גדולים ושאפתניים מאוד כמו הקמת הסכרים הגדולים בהולנד, תעלת סואץ ותעלת פנמה כדי להראות שאם נרצה אין זו אגדה. אבל לצערו של סורגל הוא נתקל גם בהתנגדות חריפה מאוד בעיקר מצד אירופאים שהיו אמורים להפסיד ממימוש התוכנית ושאותם כן היה צריך לקחת בחשבון. האיטלקים למשל לא היו מוכנים לשמוע על ייבוש הים. ערי הנמל הגדולות שלהם כמו גנואה למשל יהפכו בבת אחת ללא רלוונטיות והכלכלה האיטלקית השברירית תקרוס. גם הצרפתים שהיו עתידים לאבד את הנמל של מרסיי התנגדו. סורגל הציע לחצוב תעלות מים גדולות אל הנמלים שיהיו כעת בלב היבשת כדי לשמר את תפקודם ואפילו תכנן סכרים מיוחדים שישמרו על ונציה מוצפת, אבל הרעיונות האלה לא הפחיתו מעוצמת ההתנגדות. בתחילת שנות ה-30 תפסו הנאצים את המושכות בגרמניה וסורגל היה משוכנע שנפתח בפניו חלון הזדמנויות מעולה. הנאצים היו ידועים כחובבים של מיזמי בנייה בקנה מידה גדול לתפארת האומה הגרמנית, וסביר להניח שלא תהיה להם בעיה לדרוס באגרסיביות כל מי שיתנגד לתוכנית. לרוע מזלו של סורגל גם הנאצים לא התלהבו מאטלנטרופה. הרי הבסיס לכל התוכנית היה הרצון לאחד את אירופה, למנוע סכסוכים ומלחמות ושכולם יחיו בשלום זה עם זה. הקונספט הזה לא ממש התאים לנאצים.

Ran Levi 40:25

עם פריצת מלחמת העולם השנייה ירדה תוכנית אטלנטרופה מעל סדר היום הציבורי. סורגל המשיך לכתוב ספרים ומאות מאמרים על הנושא אבל לא זכה להצלחה מרובה. בתחילת שנות ה-50 שוב עלתה מידת ההתעניינות ברעיונותיו כחלק מהרצון לשקם את אירופה אחרי המלחמה, אבל אז הופיעה האנרגיה הגרעינית כתחליף מעשי וזול לאנרגיה הידרואלקטרית והפכה את הסכר במצר גיברלטר ללא יעיל וכך שמטה את הקרקע מתחת לרגליו של סורגל. ביום חג המולד 1952 יצא הרמן סורגל על אופניו להרצאה שהיה אמור להעביר באוניברסיטה לא הרחק מביתו. מכונית שעברה בכביש סטתה מנתיבה, פגעה בו בעוצמה וברחה מהמקום. סורגל נהרג והמכונית הפוגעת לא אותרה מעולם.

Ran Levi 41:31

הרמן סורגל הלך לעולמו לפני למעלה מ-50 שנה, אבל קשה לומר את אותו הדבר לגבי רעיונותיו. גם כיום יש מי שמזכירים את פרויקט אטלנטרופה מדי פעם ובכל שנה יוצאים מאמרים אקדמיים שמנתחים בפירוט אספקטים שונים שלו כמו יעילות כלכלית וייתכנות הנדסית. אם הייתי שואל אתכם בתחילת הפרק, עוד לפני ששמעתם על האירוע המסיני, אם ניתן לרוקן את הים התיכון כולו בתוך כמה מאות שנים, אני מניח שרובכם הייתם מטילים בכך ספק. כמה משאבות צריך לבנות כדי לרוקן ים? לאיפה נשפוך את המים? על פניו ריקון של ים שלם נראה כמו אתגר טכנולוגי בלתי אפשרי. אבל הוא אפשרי ועכשיו אתם גם יודעים איך אפשר לבצע זאת בפועל. כל מה שצריך הוא לבנות סכר גדול על מצרי גיברלטר. משימה כבירה ללא ספק אבל אחת שהייתה ניתנת ליישום כבר לפני 100 שנה בימיו של הרמן סורגל. נכון להיום השאפתנות האנושית הטבעית שלנו, זו שאפשרה לנו לבנות פירמידות ולהגיע לירח, מרוסנת על ידי הידיעה שאנחנו לא מבינים עד הסוף את המשמעויות של שינוי גיאוגרפי כל כך גדול. מי יודע אילו השלכות יהיו לייבוש של הים התיכון על האקלים העולמי? על האקולוגיה שלנו? היעלמות המים תביא להפחתת הלחץ על קליפת כדור הארץ ומי יכול לשער מה היו ההשלכות לכך? רעידות אדמה חזקות או התפרצויות געשיות בלתי צפויות? ומה יקרה אם הסכר הגדול ייפרץ בטעות? ויש לנו סיבה טובה להיות מהוססים. אבל ההיסטוריה האנושית מלמדת אותנו שקשה לנו לרסן את הדחפים הטבעיים שלנו. אם משהו אפשרי ובעל פוטנציאל מועיל, חזקה עלינו שמתישהו בעתיד יבוא מהנדס, מדען או פוליטיקאי ויציע: בואו נייבש את הים התיכון במלואו או בחלקו. לא מן הנמנע שבעוד 50, 100 או 200 שנה צאצאינו יחיו לחופיו של ים תיכון שונה מאוד מזה שאנחנו מכירים. ומה אז? מי יודע. זו יכולה להיות הצלחה מסחררת שתביא פריחה ועושר לכל העמים באזורנו, או אסון אקולוגי שכמותו קשה אפילו לדמיין. ועד שזה יקרה, תרשו לי לסיים בשיר. כולם אומרים מחר יבוא שלום, אבל עלינו כבר הקיץ הקץ. ועד שיפתחו את השסתום, אני יורד לים להתרחץ.
